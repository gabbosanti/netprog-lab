# Imposta l'azione di default per la tabella ipv4_lpm: se non c'è corrispondenza, scarta il pacchetto (drop)
table_set_default ipv4_lpm drop

################# Regole di inoltro (Forwarding): #################
# Se l'IP di destinazione è 195.11.14.5 , 195.11.14.6 , 195.11.14.7  (il client), invia al MAC 00:00:0a:00:01:01 sulla porta 2
table_add ipv4_lpm ipv4_forward 195.11.14.5/32 =>  00:00:00:00:00:02 2
table_add ipv4_lpm ipv4_forward 195.11.14.6/32 =>  00:00:00:00:00:02 2
table_add ipv4_lpm ipv4_forward 195.11.14.7/32 =>  00:00:00:00:00:02 2

# Se l'IP di destinazione è 200.1.1.7 , 200.1.1.8, 200.1.1.9 (il server), invia al MAC 00:00:00:00:00:03 sulla porta 1
table_add ipv4_lpm ipv4_forward 200.1.1.7/32 =>  00:00:0a:00:01:02 1
table_add ipv4_lpm ipv4_forward 200.1.1.8/32 =>  00:00:0a:00:02:02 1
table_add ipv4_lpm ipv4_forward 200.1.1.9/32 =>  00:00:0a:00:02:02 1

################# Configurazione della chiave AES: #################
# Scrive nei registri "keys" i segmenti della chiave crittografica (4 word da 32 bit per una chiave a 128 bit)
register_write keys 0 729683222
register_write keys 1 682545830
register_write keys 2 2885096840
register_write keys 3 164581180
# Inizializza a zero i restanti slot del registro
register_write keys 4 0
register_write keys 5 0
register_write keys 6 0
register_write keys 7 0

################## Regole di sicurezza Modbus:#################
# Applica l'azione 'cipher' (cifratura) per i pacchetti diretti al client 195.11.14.5
table_add modbus_sec cipher 195.11.14.5  =>
# Applica l'azione 'decipher' (decifratura) per i pacchetti diretti al server 200.1.1.7
table_add modbus_sec decipher 200.1.1.7 =>
EOF
